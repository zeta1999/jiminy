30  ///
31  /// \brief Structure holding the data for every sensors of a given type.
32  ///
33  /// \details Every sensors of a given type must have the same 'behavior', e.g. the same
34  /// delay interpolation order and output type. However, their physical properties
35  /// may defer, such as the delay, the noise level or the bias. This enable us to
36  /// optimize the efficiency of data storage by gathering the state of every sensor
37  /// of the given type in Eigen Vectors by simply adding an extra dimension
38  /// corresponding to the sensor ID.
39  ///
40  ///////////////////////////////////////////////////////////////////////////////////////////////
41  struct SensorSharedDataHolder_t
42  {
43  SensorSharedDataHolder_t(void) :
44  time_(),
45  data_(),
46  dataMeasured_(),
47  sensors_(),
48  num_(0),
49  delayMax_(0.0)
50  {
51  // Empty.
52  };
53 
54  ~SensorSharedDataHolder_t(void) = default;
55 
56  boost::circular_buffer_space_optimized<float64_t> time_; ///< Circular buffer of the stored timesteps
57  boost::circular_buffer_space_optimized<matrixN_t> data_; ///< Circular buffer of past sensor real data
58  matrixN_t dataMeasured_; ///< Buffer of current sensor measurement data
59  std::vector<AbstractSensorBase *> sensors_; ///< Vector of pointers to the sensors
60  int32_t num_; ///< Number of sensors of that type
61  float64_t delayMax_; ///< Maximum delay over all the sensors
62  };

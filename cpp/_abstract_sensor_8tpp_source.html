238  hresult_t AbstractSensorTpl<T>::interpolateData(void)
239  {
240  // Add STEPPER_MIN_TIMESTEP to timeDesired to avoid float comparison issues
241  float64_t const timeDesired = sharedHolder_->time_.back() - baseSensorOptions_->delay + STEPPER_MIN_TIMESTEP;
242 
243  /* Determine the position of the closest right element.
244  Bisection method can be used since times are sorted. */
245  auto bisectLeft =
246  [&](void) -> int32_t
247  {
248  int32_t left = 0;
249  int32_t right = sharedHolder_->time_.size() - 1;
250  int32_t mid = 0;
251 
252  if (timeDesired >= sharedHolder_->time_.back())
253  {
254  return right;
255  }
256  else if (timeDesired < sharedHolder_->time_.front())
257  {
258  return -1;
259  }
260 
261  while(left < right)
262  {
263  mid = (left + right) / 2;
264  if (timeDesired < sharedHolder_->time_[mid])
265  {
266  right = mid;
267  }
268  else if (timeDesired > sharedHolder_->time_[mid])
269  {
270  left = mid + 1;
271  }
272  else
273  {
274  return mid;
275  }
276  }
277 
278  if (timeDesired < sharedHolder_->time_[mid])
279  {
280  return mid - 1;
281  }
282  else
283  {
284  return mid;
285  }
286  };
287 
288  int32_t const idxLeft = bisectLeft();
289  if (timeDesired >= 0.0 && uint32_t(idxLeft + 1) < sharedHolder_->time_.size())
290  {
291  if (idxLeft < 0)
292  {
293  std::cout << "Error - AbstractSensorTpl<T>::updateDataBuffer - No data old enough is available." << std::endl;
294  return hresult_t::ERROR_GENERIC;
295  }
296  else if (baseSensorOptions_->delayInterpolationOrder == 0)
297  {
298  get() = sharedHolder_->data_[idxLeft].col(sensorIdx_);
299  }
300  else if (baseSensorOptions_->delayInterpolationOrder == 1)
301  {
302  get() = 1 / (sharedHolder_->time_[idxLeft + 1] - sharedHolder_->time_[idxLeft]) *
303  ((timeDesired - sharedHolder_->time_[idxLeft]) * sharedHolder_->data_[idxLeft + 1].col(sensorIdx_) +
304  (sharedHolder_->time_[idxLeft + 1] - timeDesired) * sharedHolder_->data_[idxLeft].col(sensorIdx_));
305  }
306  else
307  {
308  std::cout << "Error - AbstractSensorTpl<T>::updateDataBuffer - The delayInterpolationOrder must be either 0 or 1 so far." << std::endl;
309  return hresult_t::ERROR_BAD_INPUT;
310  }
311  }
312  else
313  {
314  if (baseSensorOptions_->delay > EPS)
315  {
316  // Return the oldest value since the buffer is not fully initialized yet
317  auto it = std::find_if(sharedHolder_->time_.begin(), sharedHolder_->time_.end(),
318  [] (float64_t const & t)
319  {
320  return t > 0;
321  });
322  if (it != sharedHolder_->time_.end())
323  {
324  int32_t ind = std::distance(sharedHolder_->time_.begin(), it);
325  ind = std::max(0, ind - 1);
326  get() = sharedHolder_->data_[ind].col(sensorIdx_);
327  }
328  else
329  {
330  get() = sharedHolder_->data_.back().col(sensorIdx_);
331  }
332  }
333  else
334  {
335  // Return the most recent value available
336  get() = sharedHolder_->data_.back().col(sensorIdx_);
337  }
338  }

383  {
384  // Remove some unecessary extra elements if appropriate

396  }

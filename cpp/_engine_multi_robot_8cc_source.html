702  {
703  if (forcesImpulseIt->t < STEPPER_MIN_TIMESTEP)
704  {
705  *forcesImpulseActiveIt = true;
706  }
707  }
708  }
709 
710  // Compute the internal and external forces applied on every systems
711  auto const xSplit = splitState(xCat);
712  computeAllForces(t, xSplit);
713 
714  for (auto & system : systemsDataHolder_)
715  {
716  // Get some system state proxies
717  vectorN_t const & q = system.state.q;
718  vectorN_t const & v = system.state.v;
719  vectorN_t & qDot = system.state.qDot;
720  vectorN_t & a = system.state.a;
721  vectorN_t & u = system.state.u;
722  vectorN_t & uCommand = system.state.uCommand;
723  vectorN_t & uMotor = system.state.uMotor;
724  vectorN_t & uInternal = system.state.uInternal;
725  forceVector_t & fext = system.state.fExternal;
726 
727  // Initialize the sensor data
728  system.robot->setSensorsData(t, q, v, a, uMotor);
729 
730  // Compute the actual motor effort
731  computeCommand(system, t, q, v, uCommand);
732 
733  // Compute the actual motor effort
734  system.robot->computeMotorsEfforts(t, q, v, a, uCommand);
735  uMotor = system.robot->getMotorsEfforts();
736 
737  // Compute the internal dynamics
738  computeInternalDynamics(system, t, q, v, uInternal);
739 
740  // Compute the total effort vector
741  u = uInternal;
742  for (auto const & motor : system.robot->getMotors())
743  {
744  int32_t const & motorIdx = motor->getIdx();
745  int32_t const & motorVelocityIdx = motor->getJointVelocityIdx();
746  u[motorVelocityIdx] += uMotor[motorIdx];
747  }
748 
749  // Compute dynamics
750  a = computeAcceleration(system, q, v, u, fext);
751 
752  // Project the derivative in state space
753  computePositionDerivative(system.robot->pncModel_, q, v, qDot, dt);
754 
755  // Compute the forward kinematics once again, with the updated acceleration
756  computeForwardKinematics(system, q, v, a);
757 
758  // Update the sensor data once again, with the updated effort and acceleration
759  system.robot->setSensorsData(t, q, v, a, uMotor);
760  }
761 
762  // Synchronize the global stepper state with the individual system states
763  syncStepperStateWithSystems();
764  }
765 
766  // Lock the telemetry. At this point it is no longer possible to register new variables.
767  configureTelemetry();
768 
769  // Write the header: this locks the registration of new variables
770  telemetryRecorder_->initialize(telemetryData_.get(), engineOptions_->telemetry.timeUnit);
771 
772  // Log current buffer content as first point of the log data.
773  updateTelemetry();
774 
775  // Initialize the last system states
776  for (auto & system : systemsDataHolder_)
777  {
778  system.statePrev = system.state;
779  }
780 
781  if (returnCode != hresult_t::SUCCESS)
782  {
783  stop();
784  }
785 
786  return returnCode;
787  }

789  hresult_t EngineMultiRobot::simulate(float64_t const & tEnd,
790  std::map<std::string, vectorN_t> const & xInit)
791  {
792  hresult_t returnCode = hresult_t::SUCCESS;
793 
794  if (systemsDataHolder_.empty())
795  {
796  std::cout << "Error - EngineMultiRobot::simulate - No system to simulate. Please add one before starting a simulation." << std::endl;
797  returnCode = hresult_t::ERROR_INIT_FAILED;
798  }
799 
800  if (tEnd < 5e-3)
801  {
802  std::cout << "Error - EngineMultiRobot::simulate - The duration of the simulation cannot be shorter than 5ms." << std::endl;
803  returnCode = hresult_t::ERROR_BAD_INPUT;
804  }
805 
806  // Reset the robot, controller, and engine
807  if (returnCode == hresult_t::SUCCESS)
808  {
809  returnCode = start(xInit, true, false);
810  }
811 
812  // Now that telemetry has been initialized, check simulation duration.
813  if (tEnd > telemetryRecorder_->getMaximumLogTime())
814  {
815  std::cout << "Error - EngineMultiRobot::simulate - Time overflow: with the current precision ";
816  std::cout << "the maximum value that can be logged is " << telemetryRecorder_->getMaximumLogTime();
817  std::cout << "s. Decrease logger precision to simulate for longer than that." << std::endl;
818  return hresult_t::ERROR_BAD_INPUT;
819  }
820 
821  // Integration loop based on boost::numeric::odeint::detail::integrate_times
822  while (returnCode == hresult_t::SUCCESS)
823  {
824  // Stop the simulation if the end time has been reached
825  if (tEnd - stepperState_.t < SIMULATION_MIN_TIMESTEP)
826  {
827  if (engineOptions_->stepper.verbose)
828  {
829  std::cout << "Simulation done: desired final time reached." << std::endl;
830  }
831  break;
832  }
833 
834  // Stop the simulation if any of the callbacks return false
835  bool_t isCallbackFalse = false;
836  for (auto & system : systemsDataHolder_)
837  {
838  if (!system.callbackFct(stepperState_.t, system.state.q, system.state.v))
839  {
840  isCallbackFalse = true;
841  break;
842  }
843  }
844  if (isCallbackFalse)
845  {
846  if (engineOptions_->stepper.verbose)
847  {
848  std::cout << "Simulation done: callback returned false." << std::endl;
849  }
850  break;
851  }
852 
853  // Stop the simulation if the max number of integration steps is reached
854  if (0 < engineOptions_->stepper.iterMax
855  && (uint32_t) engineOptions_->stepper.iterMax <= stepperState_.iter)
856  {
857  if (engineOptions_->stepper.verbose)
858  {
859  std::cout << "Simulation done: maximum number of integration steps exceeded." << std::endl;
860  }
861  break;
862  }
863 
864  // Perform a single integration step up to tEnd, stopping at stepperUpdatePeriod_ to log.
865  float64_t stepSize;
866  if (stepperUpdatePeriod_ > EPS)
867  {
868  stepSize = min(stepperUpdatePeriod_ , tEnd - stepperState_.t);
869  }
870  else
871  {
872  stepSize = min(engineOptions_->stepper.dtMax, tEnd - stepperState_.t);
873  }
874  returnCode = step(stepSize); // Automatic dt adjustment
875  }
876 
877  // Stop the simulation. New variables can be registered again, and the lock on the robot is released
878  stop();
879 
880  return returnCode;
881  }

883  hresult_t EngineMultiRobot::step(float64_t stepSize)
884  {
885  // Check if the simulation has started
886  if (!isSimulationRunning_)
887  {
888  std::cout << "Error - EngineMultiRobot::step - No simulation running. Please start it before using step method." << std::endl;
889  return hresult_t::ERROR_GENERIC;
890  }
891 
892  // Check if there is something wrong with the integration
893  if ((stepperState_.x.array() != stepperState_.x.array()).any()) // isnan if NOT equal to itself
894  {
895  std::cout << "Error - EngineMultiRobot::step - The low-level ode solver failed. "\
896  "Consider increasing the stepper accuracy." << std::endl;
897  return hresult_t::ERROR_GENERIC;
898  }
899 
900  // Check if the desired step size is suitable
901  if (stepSize > EPS && stepSize < SIMULATION_MIN_TIMESTEP)
902  {
903  std::cout << "Error - EngineMultiRobot::step - The requested step size is out of bounds." << std::endl;
904  return hresult_t::ERROR_BAD_INPUT;
905  }
906 
907  /* Set end time: The default step size is equal to the
908  controller update period if discrete-time, otherwise
909  it uses the sensor update period if discrete-time,
910  otherwise it uses the user-defined parameter dtMax. */
911  if (stepSize < EPS)
912  {
913  float64_t const & controllerUpdatePeriod = engineOptions_->stepper.controllerUpdatePeriod;
914  if (controllerUpdatePeriod > EPS)
915  {
916  stepSize = controllerUpdatePeriod;
917  }
918  else
919  {
920  float64_t const & sensorsUpdatePeriod = engineOptions_->stepper.sensorsUpdatePeriod;
921  if (sensorsUpdatePeriod > EPS)
922  {
923  stepSize = sensorsUpdatePeriod;
924  }
925  else
926  {
927  stepSize = engineOptions_->stepper.dtMax;
928  }
929  }
930  }
931 
932  /* Check that end time is not too large for the current
933  logging precision, otherwise abort integration. */
934  if (stepperState_.t + stepSize > telemetryRecorder_->getMaximumLogTime())
935  {
936  std::cout << "Error - EngineMultiRobot::step - Time overflow: with the current precision ";
937  std::cout << "the maximum value that can be logged is " << telemetryRecorder_->getMaximumLogTime();
938  std::cout << "s. Decrease logger precision to simulate for longer than that." << std::endl;
939  return hresult_t::ERROR_GENERIC;
940  }
941 
942  /* Avoid compounding of error using Kahan algorithm. It
943  consists in keeping track of the cumulative rounding error
944  to add it back to the sum when it gets larger than the
945  numerical precision, thus avoiding it to grows unbounded. */
946  float64_t stepSize_true = stepSize - stepperState_.tError;
947  float64_t tEnd = stepperState_.t + stepSize_true;
948  stepperState_.tError = (tEnd - stepperState_.t) - stepSize_true;
949 
950  // Get references to some internal stepper buffers
951  float64_t & t = stepperState_.t;
952  float64_t & dt = stepperState_.dt;
953  float64_t & dtLargest = stepperState_.dtLargest;
954  vectorN_t & x = stepperState_.x;
955  vectorN_t & dxdt = stepperState_.dxdt;
956 
957  // Define the stepper iterators.
958  auto systemOde =
959  [this](vectorN_t const & xIn,
960  vectorN_t & dxdtIn,
961  float64_t const & tIn)
962  {
963  this->computeSystemDynamics(tIn, xIn, dxdtIn);
964  };
965 
966  // Define a failure checker for the stepper
967  failed_step_checker fail_checker;
968 
969  /* Flag monitoring if the current time step depends of a breakpoint
970  or the integration tolerance. It will be used by the restoration
971  mechanism, if dt gets very small to reach a breakpoint, in order
972  to avoid having to perform several steps to stabilize again the
973  estimation of the optimal time step. */
974  bool_t isBreakpointReached = false;
975 
976  /* Flag monitoring if the dynamics has changed because of impulse
977  forces or the command (only in the case of discrete control).
978 
979  `try_step(rhs, x, dxdt, t, dt)` method of error controlled boost
980  steppers leverage the FSAL (first same as last) principle. It is
981  implemented by considering at the value of (x, dxdt) in argument
982  have been initialized by the user with the system dynamics at
983  current time t. Thus, if the system dynamics is discontinuous,
984  one has to manually integrate up to t-, then update dxdt to take
985  into the acceleration at t+.
986 
987  Note that ONLY the acceleration part of dxdt must be updated since
988  the projection of the velocity on the state space is not supposed
989  to have changed, and on top of that tPrev is invalid at this point
990  because it has been updated just after the last successful step.
991 
992  Note that the estimated dt is no longer very meaningful since the
993  dynamics has changed. Maybe dt should be reschedule... */
994  bool_t hasDynamicsChanged = false;
995 
996  // Start the timer used for timeout handling
997  timer_.tic();
998 
999  // Perform the integration. Do not simulate extremely small time steps.
1000  while (tEnd - t > STEPPER_MIN_TIMESTEP)
1001  {
1002  float64_t tNext = t;
1003 
1004  // Update the active set and get the next breakpoint of impulse forces
1005  float64_t tForceImpulseNext = INF;
1006  for (auto & system : systemsDataHolder_)
1007  {
1008  /* Update the active set: activate an impulse force as soon as
1009  the current time gets close enough of the application time,
1010  and deactivate it once the following the same reasoning.
1011 
1012  Note that breakpoints at the begining and the end of every
1013  impulse force at already enforced, so that the forces
1014  cannot get activated/desactivate too late. */
1015  auto forcesImpulseActiveIt = system.forcesImpulseActive.begin();
1016  auto forcesImpulseIt = system.forcesImpulse.begin();
1017  for ( ; forcesImpulseIt != system.forcesImpulse.end() ;
1018  forcesImpulseActiveIt++, forcesImpulseIt++)
1019  {
1020  float64_t const & tForceImpulse = forcesImpulseIt->t;
1021  float64_t const & dtForceImpulse = forcesImpulseIt->dt;
1022 
1023  if (t > tForceImpulse - STEPPER_MIN_TIMESTEP)
1024  {
1025  *forcesImpulseActiveIt = true;
1026  hasDynamicsChanged = true;
1027  }
1028  if (t > tForceImpulse + dtForceImpulse - STEPPER_MIN_TIMESTEP)
1029  {
1030  *forcesImpulseActiveIt = false;
1031  hasDynamicsChanged = true;
1032  }
1033  }
1034 
1035  // Update the breakpoint time iterator if necessary
1036  auto & tBreakNextIt = system.forcesImpulseBreakNextIt;
1037  if (tBreakNextIt != system.forcesImpulseBreaks.end())
1038  {
1039  if (t > *tBreakNextIt - STEPPER_MIN_TIMESTEP)
1040  {
1041  // The current breakpoint is behind in time. Switching to the next one.
1042  tBreakNextIt++;
1043  }
1044  }
1045 
1046  // Get the next breakpoint time if any
1047  if (tBreakNextIt != system.forcesImpulseBreaks.end())
1048  {
1049  tForceImpulseNext = min(tForceImpulseNext, *tBreakNextIt);
1050  }
1051  }
1052 
1053  if (stepperUpdatePeriod_ > EPS)
1054  {
1055  // Update the sensor data if necessary (only for finite update frequency)
1056  if (engineOptions_->stepper.sensorsUpdatePeriod > EPS)
1057  {
1058  float64_t const & sensorsUpdatePeriod = engineOptions_->stepper.sensorsUpdatePeriod;
1059  float64_t dtNextSensorsUpdatePeriod = sensorsUpdatePeriod - std::fmod(t, sensorsUpdatePeriod);
1060  if (dtNextSensorsUpdatePeriod < SIMULATION_MIN_TIMESTEP
1061  || sensorsUpdatePeriod - dtNextSensorsUpdatePeriod < SIMULATION_MIN_TIMESTEP)
1062  {
1063  for (auto & system : systemsDataHolder_)
1064  {
1065  vectorN_t const & q = system.state.q;
1066  vectorN_t const & v = system.state.v;
1067  vectorN_t const & a = system.state.a;
1068  vectorN_t const & uMotor = system.state.uMotor;
1069  system.robot->setSensorsData(t, q, v, a, uMotor);
1070  }
1071  }
1072  }
1073 
1074  // Update the controller command if necessary (only for finite update frequency)
1075  if (engineOptions_->stepper.controllerUpdatePeriod > EPS)
1076  {
1077  float64_t const & controllerUpdatePeriod = engineOptions_->stepper.controllerUpdatePeriod;
1078  float64_t dtNextControllerUpdatePeriod = controllerUpdatePeriod - std::fmod(t, controllerUpdatePeriod);
1079  if (dtNextControllerUpdatePeriod < SIMULATION_MIN_TIMESTEP
1080  || controllerUpdatePeriod - dtNextControllerUpdatePeriod < SIMULATION_MIN_TIMESTEP)
1081  {
1082  for (auto & system : systemsDataHolder_)
1083  {
1084  vectorN_t const & q = system.state.q;
1085  vectorN_t const & v = system.state.v;
1086  vectorN_t & uCommand = system.state.uCommand;
1087  computeCommand(system, t, q, v, uCommand);
1088  }
1089  hasDynamicsChanged = true;
1090  }
1091  }
1092  }
1093 
1094  // Fix the FSAL issue if the dynamics has changed
1095  if (hasDynamicsChanged)
1096  {
1097  computeSystemDynamics(t, x, dxdt);
1098  syncSystemsStateWithStepper();
1099  }
1100 
1101  if (stepperUpdatePeriod_ > EPS)
1102  {
1103  /* Get the time of the next breakpoint for the ODE solver:
1104  a breakpoint occurs if we reached tEnd, if an external force
1105  is applied, or if we need to update the sensors / controller. */
1106  float64_t dtNextGlobal; // dt to apply for the next stepper step because of the various breakpoints
1107  float64_t dtNextUpdatePeriod = stepperUpdatePeriod_ - std::fmod(t, stepperUpdatePeriod_);
1108  if (dtNextUpdatePeriod < SIMULATION_MIN_TIMESTEP)
1109  {
1110  /* Step to reach next sensors/controller update is too short:
1111  skip one controller update and jump to the next one.
1112  Note that in this case, the sensors have already been
1113  updated in anticipation in previous loop. */
1114  dtNextGlobal = min(dtNextUpdatePeriod + stepperUpdatePeriod_,
1115  tForceImpulseNext - t);
1116  }
1117  else
1118  {
1119  dtNextGlobal = min(dtNextUpdatePeriod, tForceImpulseNext - t);
1120  }
1121 
1122  /* Check if the next dt to about equal to the time difference
1123  between the current time (it can only be smaller) and
1124  enforce next dt to exactly match this value in such a case. */
1125  if (tEnd - t - STEPPER_MIN_TIMESTEP < dtNextGlobal)
1126  {
1127  dtNextGlobal = tEnd - t;
1128  }
1129  tNext += dtNextGlobal;
1130 
1131  // Compute the next step using adaptive step method
1132  while (tNext - t > EPS)
1133  {
1134  /* Adjust stepsize to end up exactly at the next breakpoint,
1135  prevent steps larger than dtMax, trying to reach multiples of
1136  STEPPER_MIN_TIMESTEP whenever possible. The idea here is to
1137  reach only multiples of 1us, making logging easier, given that,
1138  in robotics, 1us can be consider an 'infinitesimal' time. This
1139  arbitrary threshold many not be suited for simulating different,
1140  faster dynamics, that require sub-microsecond precision. */
1141  dt = min(dt, tNext - t, engineOptions_->stepper.dtMax);
1142  if (tNext - (t + dt) < STEPPER_MIN_TIMESTEP)
1143  {
1144  dt = tNext - t;
1145  }
1146  if (dt > SIMULATION_MIN_TIMESTEP)
1147  {
1148  float64_t const dtResidual = std::fmod(dt, SIMULATION_MIN_TIMESTEP);
1149  if (dtResidual > STEPPER_MIN_TIMESTEP
1150  && dtResidual < SIMULATION_MIN_TIMESTEP - STEPPER_MIN_TIMESTEP
1151  && dt - dtResidual > STEPPER_MIN_TIMESTEP)
1152  {
1153  dt -= dtResidual;
1154  }
1155  }
1156 
1157  /* Break the loop if dt is getting too small.
1158  Don't worry, an exception will be raised later. */
1159  if (dt < STEPPER_MIN_TIMESTEP)
1160  {
1161  break;
1162  }
1163 
1164  /* Break the loop in case of timeout.
1165  Don't worry, an exception will be raised later. */
1166  timer_.toc();
1167  if (EPS < engineOptions_->stepper.timeout
1168  && engineOptions_->stepper.timeout < timer_.dt)
1169  {
1170  break;
1171  }
1172 
1173  /* A breakpoint has been reached dt has been decreased
1174  wrt the largest possible dt within integration tol. */
1175  isBreakpointReached = (stepperState_.dtLargest > dt);
1176 
1177  // Set the timestep to be tried by the stepper
1178  dtLargest = dt;
1179 
1180  if (try_step(stepper_, systemOde, x, dxdt, t, dtLargest))
1181  {
1182  // Reset the fail counter
1183  fail_checker.reset();
1184 
1185  // Synchronize the individual system states
1186  syncSystemsStateWithStepper();
1187 
1188  // Increment the iteration counter only for successful steps
1189  stepperState_.iter++;
1190 
1191  // Log every stepper state only if the user asked for
1192  if (engineOptions_->stepper.logInternalStepperSteps)
1193  {
1194  updateTelemetry();
1195  }
1196 
1197  /* Restore the step size dt if it has been significantly
1198  decreased to because of a breakpoint. It is set
1199  equal to the last available largest dt to be known,
1200  namely the second to last successfull step. */
1201  if (isBreakpointReached)
1202  {
1203  /* Restore the step size if and only if:
1204  - the next estimated largest step size is larger than
1205  the requested one for the current (successful) step.
1206  - the next estimated largest step size is significantly
1207  smaller than the estimated largest step size for the
1208  previous step. */
1209  float64_t dtRestoreThresholdAbs = stepperState_.dtLargestPrev *
1210  engineOptions_->stepper.dtRestoreThresholdRel;
1211  if (dt < dtLargest && dtLargest < dtRestoreThresholdAbs)
1212  {
1213  dtLargest = stepperState_.dtLargestPrev;
1214  }
1215  }
1216 
1217  /* Backup the stepper and systems' state on success only:
1218  - t at last successful iteration is used to compute dt,
1219  which is project the accelation in the state space
1220  instead of SO3^2.
1221  - dtLargestPrev is used to restore the largest step
1222  size in case of a breakpoint requiring lowering it.
1223  - the acceleration and effort at the last successful
1224  iteration is used to update the sensors' data in
1225  case of continuous sensing. */
1226  stepperState_.tPrev = t;
1227  stepperState_.dtLargestPrev = stepperState_.dtLargest;
1228  for (auto & system : systemsDataHolder_)
1229  {
1230  system.statePrev = system.state;
1231  }
1232  }
1233  else
1234  {
1235  /* Check for possible overflow of failed steps
1236  in step size adjustment. */
1237  fail_checker();
1238 
1239  // Increment the failed iteration counter
1240  stepperState_.iterFailed++;
1241  }
1242 
1243  // Initialize the next dt
1244  dt = dtLargest;
1245  }
1246  }
1247  else
1248  {
1249  /* Make sure it ends exactly at the tEnd, never exceeds
1250  dtMax, and stop to apply impulse forces. */
1251  dt = min(dt,
1252  engineOptions_->stepper.dtMax,
1253  tEnd - t,
1254  tForceImpulseNext - t);
1255 
1256  /* A breakpoint has been reached dt has been decreased
1257  wrt the largest possible dt within integration tol. */
1258  isBreakpointReached = (stepperState_.dtLargest > dt);
1259 
1260  // Compute the next step using adaptive step method
1261  bool_t isStepSuccessful = false;
1262  while (!isStepSuccessful)
1263  {
1264  // Set the timestep to be tried by the stepper
1265  dtLargest = dt;
1266 
1267  // Try to do a step
1268  isStepSuccessful = try_step(stepper_, systemOde, x, dxdt, t, dtLargest);
1269 
1270  if (isStepSuccessful)
1271  {
1272  // Reset the fail counter
1273  fail_checker.reset();
1274 
1275  // Synchronize the individual system states
1276  syncSystemsStateWithStepper();
1277 
1278  // Increment the iteration counter
1279  stepperState_.iter++;
1280 
1281  // Log every stepper state only if required
1282  if (engineOptions_->stepper.logInternalStepperSteps)
1283  {
1284  updateTelemetry();
1285  }
1286 
1287  // Restore the step size if necessary
1288  if (isBreakpointReached)
1289  {
1290  float64_t dtRestoreThresholdAbs = stepperState_.dtLargestPrev *
1291  engineOptions_->stepper.dtRestoreThresholdRel;
1292  if (dt < dtLargest && dtLargest < dtRestoreThresholdAbs)
1293  {
1294  dtLargest = stepperState_.dtLargestPrev;
1295  }
1296  }
1297 
1298  // Backup the stepper and systems' state
1299  stepperState_.tPrev = t;
1300  stepperState_.dtLargestPrev = dtLargest;
1301  for (auto & system : systemsDataHolder_)
1302  {
1303  system.statePrev = system.state;
1304  }
1305  }
1306  else
1307  {
1308  /* check for possible overflow of failed steps
1309  in step size adjustment. */
1310  fail_checker();
1311 
1312  // Increment the failed iteration counter
1313  stepperState_.iterFailed++;
1314  }
1315 
1316  // Initialize the next dt
1317  dt = dtLargest;
1318  }
1319  }
1320 
1321  if (dt < STEPPER_MIN_TIMESTEP)
1322  {
1323  std::cout << "Error - EngineMultiRobot::step - The internal time step is getting too small. "\
1324  "Impossible to integrate physics further in time." << std::endl;
1325  return hresult_t::ERROR_GENERIC;
1326  }
1327 
1328  timer_.toc();
1329  if (EPS < engineOptions_->stepper.timeout
1330  && engineOptions_->stepper.timeout < timer_.dt)
1331  {
1332  std::cout << "Error - EngineMultiRobot::step - Step computation timeout." << std::endl;
1333  return hresult_t::ERROR_GENERIC;
1334  }
1335  }
1336 
1337  /* Update the final time and dt to make sure it corresponds
1338  to the desired values and avoid compounding of error.
1339  Anyway the user asked for a step of exactly stepSize,
1340  so he is expecting this value to be reached. */
1341  stepperState_.t = tEnd;
1342  stepperState_.dt = stepSize;
1343

1741  pinocchio::Force EngineMultiRobot::computeContactDynamics(systemDataHolder_t const & system,
1742  int32_t const & frameIdx) const
1743  {
1744  /* Returns the external force in the contact frame.
1745  It must then be converted into a force onto the parent joint.
1746  /!\ Note that the contact dynamics depends only on kinematics data. /!\ */
1747 
1748  contactOptions_t const & contactOptions_ = engineOptions_->contacts;
1749 
1750  matrix3_t const & tformFrameRot = system.robot->pncData_.oMf[frameIdx].rotation();
1751  vector3_t const & posFrame = system.robot->pncData_.oMf[frameIdx].translation();
1752 
1753  // Initialize the contact force
1754  vector3_t fextInWorld;
1755  auto ground = engineOptions_->world.groundProfile(posFrame);
1756  float64_t const & zGround = std::get<float64_t>(ground);
1757  vector3_t & nGround = std::get<vector3_t>(ground);
1758  nGround.normalize();
1759  float64_t const depth = (posFrame(2) - zGround) * nGround(2); // First-order projection (exact assuming flat surface)
1760 
1761  if (depth < 0.0)
1762  {
1763  // Get frame motion in the motion frame.
1764  vector3_t const motionFrame = pinocchio::getFrameVelocity(
1765  system.robot->pncModel_, system.robot->pncData_, frameIdx).linear();
1766  vector3_t const vFrameInWorld = tformFrameRot * motionFrame;
1767  float64_t const vDepth = vFrameInWorld.dot(nGround);
1768 
1769  // Compute normal force
1770  float64_t fextNormal = 0.0;
1771  if (vDepth < 0.0)
1772  {
1773  fextNormal -= contactOptions_.damping * vDepth;
1774  }
1775  fextNormal -= contactOptions_.stiffness * depth;
1776  fextInWorld = fextNormal * nGround;
1777 
1778  // Compute friction forces
1779  vector3_t const vTangential = vFrameInWorld - vDepth * nGround;
1780  float64_t const vNorm = vTangential.norm();
1781 
1782  float64_t frictionCoeff = 0.0;
1783  if (vNorm > contactOptions_.frictionStictionVel)
1784  {
1785  if (vNorm < (1.0 + contactOptions_.frictionStictionRatio) * contactOptions_.frictionStictionVel)
1786  {
1787  float64_t const vRatio = vNorm / contactOptions_.frictionStictionVel;
1788  frictionCoeff = (contactOptions_.frictionDry * ((1.0 + contactOptions_.frictionStictionRatio) - vRatio)
1789  - contactOptions_.frictionViscous * (1.0 - vRatio)) / contactOptions_.frictionStictionRatio;
1790  }
1791  else
1792  {
1793  frictionCoeff = contactOptions_.frictionViscous;
1794  }
1795  }
1796  else
1797  {
1798  float64_t const vRatio = vNorm / contactOptions_.frictionStictionVel;
1799  frictionCoeff = contactOptions_.frictionDry * vRatio;
1800  }
1801  float64_t const fextTangential = frictionCoeff * fextNormal;
1802  fextInWorld += -fextTangential * vTangential;
1803 
1804  // Add blending factor
1805  if (contactOptions_.transitionEps > EPS)
1806  {
1807  float64_t const blendingFactor = -depth / contactOptions_.transitionEps;
1808  float64_t const blendingLaw = std::tanh(2 * blendingFactor);
1809  fextInWorld *= blendingLaw;
1810  }
1811  }
1812  else
1813  {
1814  fextInWorld.setZero();
1815  }
1816 
1817  return {fextInWorld, vector3_t::Zero()};
1818  }

1837  vectorN_t & u) const

1840  u.setZero();

1843  system.controller->internalDynamics(t, q, v, u);
1844 
1845  // Define some proxies
1846  auto const & jointOptions = engineOptions_->joints;
1847  pinocchio::Model const & pncModel = system.robot->pncModel_;
1848 
1849  // Enforce the position limit for the rigid joints only (TODO: Add support of spherical and planar joints)
1850  if (system.robot->mdlOptions_->joints.enablePositionLimit)
1851  {
1852  vectorN_t const & positionLimitMin = system.robot->getPositionLimitMin();
1853  vectorN_t const & positionLimitMax = system.robot->getPositionLimitMax();
1854  for (int32_t const & rigidIdx : system.robot->getRigidJointsModelIdx())
1855  {
1856  uint32_t const & positionIdx = pncModel.joints[rigidIdx].idx_q();
1857  uint32_t const & velocityIdx = pncModel.joints[rigidIdx].idx_v();
1858  int32_t const & jointDof = pncModel.joints[rigidIdx].nq();
1859  for (int32_t j = 0; j < jointDof; j++)
1860  {
1861  float64_t const & qJoint = q[positionIdx + j];
1862  float64_t const & vJoint = v[velocityIdx + j];
1863  float64_t const & qJointMin = positionLimitMin[positionIdx + j];
1864  float64_t const & qJointMax = positionLimitMax[positionIdx + j];
1865

1867  float64_t vJointError = 0.0;
1868  if (qJoint > qJointMax)
1869  {
1870  qJointError = qJoint - qJointMax;
1871  vJointError = std::max(vJoint, 0.0);
1872  }
1873  else if (qJoint < qJointMin)
1874  {
1875  qJointError = qJoint - qJointMin;
1876  vJointError = std::min(vJoint, 0.0);
1877  }
1878  float64_t const blendingFactor = std::abs(qJointError - jointOptions.transitionPositionEps *
1879  std::tanh(qJointError / jointOptions.transitionPositionEps));
1880  float64_t const forceJoint = - jointOptions.boundStiffness * qJointError
1881  - jointOptions.boundDamping * blendingFactor * vJointError;
1882 
1883  u[velocityIdx + j] += forceJoint;
1884  }
1885  }
1886  }
1887 
1888  // Enforce the velocity limit (do not support spherical joints)
1889  if (system.robot->mdlOptions_->joints.enableVelocityLimit)
1890  {
1891  vectorN_t const & velocityLimitMax = system.robot->getVelocityLimit();
1892  for (int32_t const & rigidIdx : system.robot->getRigidJointsModelIdx())
1893  {
1894  uint32_t const & velocityIdx = pncModel.joints[rigidIdx].idx_v();
1895  uint32_t const & jointDof = pncModel.joints[rigidIdx].nq();
1896  for (uint32_t j = 0; j < jointDof; j++)
1897  {
1898  float64_t const & vJoint = v[velocityIdx + j];
1899  float64_t const & vJointMin = -velocityLimitMax[velocityIdx + j];
1900  float64_t const & vJointMax = velocityLimitMax[velocityIdx + j];
1901 
1902  float64_t vJointError = 0.0;
1903  if (vJoint > vJointMax)
1904  {
1905  vJointError = vJoint - vJointMax;
1906  }
1907  else if (vJoint < vJointMin)
1908  {
1909  vJointError = vJoint - vJointMin;

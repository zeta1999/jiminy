580  hresult_t Robot::refreshConstraintsProxies(void)
581  {
582  hresult_t returnCode = hresult_t::SUCCESS;
583  vectorN_t q = pinocchio::neutral(pncModel_);
584  vectorN_t v = vectorN_t::Zero(pncModel_.nv);
585 
586  // Resize zeroAccelerationVector_ to the right size
587  zeroAccelerationVector_ = vectorN_t::Zero(pncModel_.nv);
588 
589  int constraintSize = 0;
590  for (auto & constraint : constraintsHolder_)
591  {
592  if (returnCode == hresult_t::SUCCESS)
593  {
594  returnCode = constraint.constraint_->refreshProxies();
595  }
596  if (returnCode == hresult_t::SUCCESS)
597  {
598  // Call constraint on neutral position and zero velocity.
599  matrixN_t J = constraint.constraint_->getJacobian(q);
600  vectorN_t drift = constraint.constraint_->getDrift(q, v);
601 
602  // Verify dimensions.
603  if (J.cols() != pncModel_.nv)
604  {
605  std::cout << "Error - Robot::refreshConstraintsProxies: constraint "\
606  "has an invalid jacobian (wrong number of columns)." << std::endl;
607  returnCode = hresult_t::ERROR_GENERIC;
608  }
609  if (drift.size() != J.rows())
610  {
611  std::cout << "Error - Robot::refreshConstraintsProxies: constraint "\
612  "has inconsistent jacobian and drift (size mismatch)." << std::endl;
613  returnCode = hresult_t::ERROR_GENERIC;
614  }
615  if (returnCode == hresult_t::SUCCESS)
616  {
617  // Store constraint size.
618  constraint.dim_ = J.rows();
619  constraintSize += J.rows();
620  }
621  }
622  }
623 
624  // Reset jacobian and drift to 0.
625  if (returnCode == hresult_t::SUCCESS)
626  {
627  constraintsJacobian_ = matrixN_t::Zero(constraintSize, pncModel_.nv);
628  constraintsDrift_ = vectorN_t::Zero(constraintSize);
629  }
630  return returnCode;
631  }

1339  void Robot::computeConstraints(Eigen::Ref<vectorN_t const> const & q,
1340  Eigen::Ref<vectorN_t const> const & v)
1341  {
1342  // Compute joint jacobian.
1343  pinocchio::computeJointJacobians(pncModel_, pncData_, q);
1344  pinocchio::forwardKinematics(pncModel_, pncData_, q, v, zeroAccelerationVector_);
1345 
1346  uint32_t currentRow = 0;
1347  for (auto & constraint : constraintsHolder_)
1348  {
1349  matrixN_t J = constraint.constraint_->getJacobian(q);
1350  vectorN_t drift = constraint.constraint_->getDrift(q, v);
1351 
1352  uint32_t constraintDim = J.rows();
1353  // Resize matrix if needed.
1354  if (constraintDim != constraint.dim_)
1355  {
1356  constraintsJacobian_.conservativeResize(
1357  constraintsJacobian_.rows() + constraintDim - constraint.dim_,
1358  Eigen::NoChange);
1359  constraintsDrift_.conservativeResize(constraintsDrift_.size() + constraintDim - constraint.dim_);
1360  constraint.dim_ = constraintDim;
1361  }
1362  constraintsJacobian_.block(currentRow, 0, constraintDim, pncModel_.nv) = J;
1363  constraintsDrift_.segment(currentRow, constraintDim) = drift;
1364  currentRow += constraintDim;
1365  }

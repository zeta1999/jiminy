116  hresult_t TelemetryRecorder::createNewChunk()
117  {
118  hresult_t returnCode = hresult_t::SUCCESS;
119 
120  // Close the current MemoryDevice, if any and if it was opened
121  if (!flows_.empty())
122  {
123  flows_.back().close();
124  }
125 
126  /* Create a new chunk.
127  The size of the first chunk is chosen to be large enough
128  to contain the whole header (with constants). Doing this
129  does not really affect the performances since it is written
130  only once, at init of the simulation. The optimized buffer
131  size is used for the log data. */
132  uint32_t isHeaderThere = flows_.empty();
133  uint32_t maxBufferSize = std::max(TELEMETRY_MAX_BUFFER_SIZE, isHeaderThere * headerSize_);
134  uint32_t maxRecordedDataLines = ((maxBufferSize - isHeaderThere * headerSize_) / recordedBytesDataLine_);
135  recordedBytesLimits_ = isHeaderThere * headerSize_ + maxRecordedDataLines * recordedBytesDataLine_;
136  flows_.emplace_back(recordedBytesLimits_);
137  returnCode = flows_.back().open(OpenMode::READ_WRITE);
138 
139  if (returnCode == hresult_t::SUCCESS)
140  {
141  recordedBytes_ = 0;
142  }
143 
144  return returnCode;
145  }

147  hresult_t TelemetryRecorder::flushDataSnapshot(float64_t const & timestamp)
148  {
149  hresult_t returnCode = hresult_t::SUCCESS;
150 
151  if (recordedBytes_ == recordedBytesLimits_)
152  {
153  returnCode = createNewChunk();
154  }
155 
156  if (returnCode == hresult_t::SUCCESS)
157  {
158  // Write new line token
159  flows_.back().write(START_LINE_TOKEN);
160 
161  // Write time
162  flows_.back().write(static_cast<int32_t>(std::round(timestamp * timeLoggingPrecision_)));
163 
164  // Write data, integers first
165  flows_.back().write(reinterpret_cast<uint8_t const*>(integersAddress_), integerSectionSize_);
166 
167  // Write data, floats last
168  flows_.back().write(reinterpret_cast<uint8_t const*>(floatsAddress_), floatSectionSize_);
169 
170  // Update internal counter
171  recordedBytes_ += recordedBytesDataLine_;
172  }
173 
174  return returnCode;

206  void TelemetryRecorder::getData(std::vector<std::string> & header,
207  std::vector<float64_t> & timestamps,
208  std::vector<std::vector<int32_t> > & intData,
209  std::vector<std::vector<float32_t> > & floatData,
210  std::vector<AbstractIODevice *> & flows,
211  int64_t const & integerSectionSize,
212  int64_t const & floatSectionSize,
213  int64_t const & headerSize,
214  int64_t recordedBytesDataLine)
215  {
216  header.clear();
217  timestamps.clear();
218  intData.clear();
219  floatData.clear();
220 
221  if (!flows.empty())
222  {
223  int32_t timestamp;
224  std::vector<int32_t> intDataLine;
225  intDataLine.resize(integerSectionSize / sizeof(int32_t));
226  std::vector<float32_t> floatDataLine;
227  floatDataLine.resize(floatSectionSize / sizeof(float32_t));
228 
229  bool_t isReadingHeaderDone = false;
230  for (auto & flow : flows)
231  {
232  // Save the cursor position and move it to the beginning
233  int64_t const pos_old = flow->pos();
234  flow->seek(0);
235 
236  // Dealing with version flag, constants, header, and descriptor
237  if (!isReadingHeaderDone)
238  {
239  int64_t header_version_length = sizeof(int32_t);
240  flow->seek(header_version_length); // Skip the version flag
241  std::vector<char_t> headerCharBuffer;
242  headerCharBuffer.resize(headerSize - header_version_length);
243  flow->read(headerCharBuffer);
244  char_t const * pHeader = &headerCharBuffer[0];
245  uint32_t posHeader = 0;
246  std::string fieldHeader(pHeader);
247  while (true)
248  {
249  header.emplace_back(std::move(fieldHeader));
250  posHeader += header.back().size() + 1;
251  fieldHeader = std::string(pHeader + posHeader);
252  if (fieldHeader.size() == 0 || posHeader >= headerCharBuffer.size())
253  {
254  break;
255  }
256  if (posHeader + fieldHeader.size() > headerCharBuffer.size())
257  {
258  fieldHeader = std::string(pHeader + posHeader, headerCharBuffer.size() - posHeader);
259  header.emplace_back(std::move(fieldHeader));
260  break;
261  }
262  }
263  isReadingHeaderDone = true;
264  }
265 
266  // In header, look for timeUnit constant - if not found, use default time unit.
267  float64_t timeUnit = TELEMETRY_DEFAULT_TIME_UNIT;
268  auto const lastConstantIt = std::find(header.begin(), header.end(), START_COLUMNS);
269  for (auto constantIt = header.begin() ; constantIt != lastConstantIt ; constantIt++)
270  {
271  int32_t const delimiter = constantIt->find("=");
272  if (constantIt->substr(0, delimiter) == TIME_UNIT)
273  {
274  timeUnit = std::stof(constantIt->substr(delimiter + 1));
275  break;
276  }
277  }
278 
279  // Dealing with data lines, starting with new line flag, time, integers, and ultimately floats
280  if (recordedBytesDataLine > 0)
281  {
282  uint32_t numberLines = (flow->size() - flow->pos()) / recordedBytesDataLine;
283  timestamps.reserve(timestamps.size() + numberLines);
284  intData.reserve(intData.size() + numberLines);
285  floatData.reserve(floatData.size() + numberLines);
286  }
287 
288  while (flow->bytesAvailable() > 0)
289  {
290  flow->seek(flow->pos() + START_LINE_TOKEN.size()); // Skip new line flag
291  flow->readData(&timestamp, sizeof(int32_t));
292  flow->readData(intDataLine.data(), integerSectionSize);
293  flow->readData(floatDataLine.data(), floatSectionSize);
294 
295  if (!timestamps.empty() && timestamp == 0)
296  {
297  // The buffer is not full, must stop reading !
298  break;
299  }
300 
301  timestamps.emplace_back(static_cast<float64_t>(timestamp / timeUnit));
302  intData.emplace_back(intDataLine);
303  floatData.emplace_back(floatDataLine);
304  }
305 
306  // Restore the cursor position
307  flow->seek(pos_old);
308  }
309  }
310  }

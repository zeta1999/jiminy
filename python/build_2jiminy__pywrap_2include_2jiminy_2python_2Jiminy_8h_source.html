465  std::string const & sensorType)
466  {
467  try
468  {
469  auto & sensorsDataType = self.at(sensorType);
470  auto sensorDataIt = sensorsDataType.begin();
471 
472  npy_intp dims[2] = {npy_intp(sensorDataIt->value.size()),
473  npy_intp(sensorsDataType.size())};
474  PyObject * dataPyPtr = PyArray_SimpleNew(2, dims, NPY_FLOAT64);

476  PyArray_DATA(reinterpret_cast<PyArrayObject *>(dataPyPtr)));
477 
478  for (; sensorDataIt != sensorsDataType.end(); sensorDataIt++)
479  {
480  for (uint32_t i=0 ; i < dims[0] ; i++)
481  {
482  auto dataIdxOffset = i * dims[1] + sensorDataIt->idx;
483  *(dataPyDataPtr + dataIdxOffset) = sensorDataIt->value[i];
484  }
485  }

1782  static bp::tuple formatLog(std::vector<std::string> const & header,
1783  std::vector<float64_t> & timestamps,
1784  std::vector<std::vector<int32_t> > & intData,
1785  std::vector<std::vector<float32_t> > & floatData,
1786  bool_t const & clear_memory = true)
1787  {
1788  bp::dict constants;
1789  bp::dict data;
1790 
1791  // Get constants
1792  int32_t const lastConstantIdx = std::distance(
1793  header.begin(), std::find(header.begin(), header.end(), START_COLUMNS));
1794  for (int32_t i = 1; i < lastConstantIdx; i++)
1795  {
1796  int32_t const delimiter = header[i].find("=");
1797  constants[header[i].substr(0, delimiter)] = header[i].substr(delimiter + 1);
1798  }
1799 
1800  // Get Global.Time
1801  if (!timestamps.empty())
1802  {
1803  Eigen::Ref<vectorN_t> timeBuffer = vectorN_t::Map(
1804  timestamps.data(), timestamps.size());
1805  PyObject * valuePyTime(getNumpyReference(timeBuffer));
1806  data[header[lastConstantIdx + 1]] = bp::object(bp::handle<>(
1807  PyArray_FROM_OF(valuePyTime, NPY_ARRAY_ENSURECOPY)));
1808  Py_XDECREF(valuePyTime);
1809  }
1810 
1811  // Get intergers
1812  Eigen::Matrix<int32_t, Eigen::Dynamic, Eigen::Dynamic> intMatrix;
1813  if (!intData.empty())
1814  {
1815  intMatrix.resize(timestamps.size(), intData[0].size());
1816  for (uint32_t i=0; i<intData.size(); i++)
1817  {
1818  intMatrix.row(i) = Eigen::Matrix<int32_t, 1, Eigen::Dynamic>::Map(
1819  intData[i].data(), intData[0].size());
1820  }
1821  if (clear_memory)
1822  {
1823  intData.clear();
1824  }
1825 
1826  for (uint32_t i=0; i<intMatrix.cols(); i++)
1827  {
1828  Eigen::Ref<Eigen::Matrix<int32_t, -1, 1> > intCol(intMatrix.col(i));
1829  PyObject * valuePyInt(getNumpyReference(intCol));
1830  std::string const & header_i = header[i + (lastConstantIdx + 1) + 1];
1831  /* One must make copies with PyArray_FROM_OF instead of using
1832  raw pointer for floatMatrix and setting NPY_ARRAY_OWNDATA
1833  because otherwise Python is not able to free the memory
1834  associated with each columns independently. Moreover, one
1835  must decrease manually the counter reference for some reasons... */
1836  data[header_i] = bp::object(bp::handle<>(
1837  PyArray_FROM_OF(valuePyInt, NPY_ARRAY_ENSURECOPY)));
1838  Py_XDECREF(valuePyInt);
1839  }
1840  }
1841 
1842  // Get floats
1843  Eigen::Matrix<float32_t, Eigen::Dynamic, Eigen::Dynamic> floatMatrix;
1844  if (!floatData.empty())
1845  {
1846  floatMatrix.resize(timestamps.size(), floatData[0].size());
1847  for (uint32_t i=0; i<floatData.size(); i++)
1848  {
1849  floatMatrix.row(i) = Eigen::Matrix<float32_t, 1, Eigen::Dynamic>::Map(
1850  floatData[i].data(), floatData[0].size());
1851  }
1852  if (clear_memory)
1853  {
1854  floatData.clear();
1855  }
1856 
1857  for (uint32_t i=0; i<floatMatrix.cols(); i++)
1858  {
1859  Eigen::Ref<Eigen::Matrix<float32_t, -1, 1> > floatCol(floatMatrix.col(i));
1860  PyObject * valuePyFloat(getNumpyReference(floatCol));
1861  std::string const & header_i =
1862  header[i + (lastConstantIdx + 1) + 1 + intData[0].size()];
1863  data[header_i] = bp::object(bp::handle<>(
1864  PyArray_FROM_OF(valuePyFloat, NPY_ARRAY_ENSURECOPY)));
1865  Py_XDECREF(valuePyFloat);
1866  }
1867  }

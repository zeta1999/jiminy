30  """
31  # Transformation Matrix corresponding to a rotation about x,y or z

33  """

45  def matrixToRpy(M):
46  """
47  # Convert from Transformation Matrix to Roll, Pitch, Yaw
48  """
49  m = sqrt(M[2, 1] ** 2 + M[2, 2] ** 2)
50  p = atan2(-M[2, 0], m)
51 
52  if abs(abs(p) - pi / 2) < 0.001:
53  r = 0
54  y = -atan2(M[0, 1], M[1, 1])
55  else:
56  y = atan2(M[1, 0], M[0, 0]) # alpha
57  r = atan2(M[2, 1], M[2, 2]) # gamma
58 
59  return np.array([r, p, y], dtype=np.float64)

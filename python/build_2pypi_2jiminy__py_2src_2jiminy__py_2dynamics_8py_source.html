27 def update_quantities(robot,
28  position,
29  velocity=None,
30  acceleration=None,
31  update_physics=True,
32  update_com=False,
33  update_energy=False,
34  update_jacobian=False,
35  use_theoretical_model=True):
36  """
37  @brief Compute all quantities using position, velocity and acceleration
38  configurations.
39 
40  @details Run multiple algorithms to compute all quantities
41  which can be known with the model position, velocity
42  and acceleration configuration.
43 
44  This includes:
45  - body spatial transforms,
46  - body spatial velocities,
47  - body spatial drifts,
48  - body transform acceleration,
49  - body transform jacobians,
50  - center-of-mass position,
51  - center-of-mass velocity,
52  - center-of-mass drift,
53  - center-of-mass acceleration,
54  (- center-of-mass jacobian : No Python binding available so far),
55  - articular inertia matrix,
56  - non-linear effects (Coriolis + gravity)
57 
58  Computation results are stored in internal
59  data and can be retrieved with associated getters.
60 
61  @note This function modifies internal data.
62 
63  @param robot The jiminy robot
64  @param position Joint position vector
65  @param velocity Joint velocity vector
66  @param acceleration Joint acceleration vector
67  """
68  if use_theoretical_model:
69  pnc_model = robot.pinocchio_model_th
70  pnc_data = robot.pinocchio_data_th
71  else:
72  pnc_model = robot.pinocchio_model
73  pnc_data = robot.pinocchio_data
74 
75  if (update_physics and update_com and \
76  update_energy and update_jacobian and \
77  velocity is not None):
78  pin.computeAllTerms(pnc_model, pnc_data, position, velocity)
79  else:
80  if update_physics:
81  if velocity is not None:
82  pin.nonLinearEffects(pnc_model, pnc_data, position, velocity)
83  pin.crba(pnc_model, pnc_data, position)
84 
85  if update_jacobian:
86  # if update_com:
87  # pin.getJacobianComFromCrba(pnc_model, pnc_data)
88  pin.computeJointJacobians(pnc_model, pnc_data)
89 
90  if update_com:
91  if velocity is None:
92  pin.centerOfMass(pnc_model, pnc_data, position)
93  elif acceleration is None:
94  pin.centerOfMass(pnc_model, pnc_data, position, velocity)
95  else:
96  pin.centerOfMass(pnc_model, pnc_data, position, velocity, acceleration)
97  else:
98  if velocity is None:
99  pin.forwardKinematics(pnc_model, pnc_data, position)
100  elif acceleration is None:
101  pin.forwardKinematics(pnc_model, pnc_data, position, velocity)
102  else:
103  pin.forwardKinematics(pnc_model, pnc_data, position, velocity, acceleration)
104  pin.framesForwardKinematics(pnc_model, pnc_data, position)
105 
106  if update_energy:
107  if velocity is not None:
108  pin.kineticEnergy(pnc_model, pnc_data, position, velocity, False)
109  pin.potentialEnergy(pnc_model, pnc_data, position, False)

111 def get_body_index_and_fixedness(robot, body_name, use_theoretical_model=True):
112  """
113  @brie Retrieve a body index in model and its fixedness from the body name.
114 
115  @details The method searchs bodyNames array and fixedBodyNames array
116  for the provided bodyName.
117 
118  @param robot The jiminy robot
119  @param body_name The name of the body for which index and fixedness are needed
120  @return body_id The index of the body whose name is bodyName in model
121  @return is_body_fixed Whether or not the body is considered as a fixed-body by the model
122  """
123  if use_theoretical_model:
124  pnc_model = robot.pinocchio_model_th
125  else:
126  pnc_model = robot.pinocchio_model
127 
128  frame_id = pnc_model.getFrameId(body_name)
129  parent_frame_id = pnc_model.frames[frame_id].previousFrame
130  parent_frame_type = pnc_model.frames[parent_frame_id].type
131  is_body_fixed = (parent_frame_type == pin.FrameType.FIXED_JOINT)
132  if is_body_fixed:
133  body_id = frame_id
134  else:
135  body_id = pnc_model.frames[frame_id].parent
136 
137  return body_id, is_body_fixed

235 def _compute_closest_contact_frame(robot, ground_profile=None, use_theoretical_model=True):
236  """
237  @brief Compute the closest contact point to the ground, in their respective local
238  frame and wrt the ground position and orientation.
239 
240  @details This method can be used in conjunction with compute_freeflyer_state_from_fixed_body
241  to determine the right fixed_body_name that ensures no contact points are going
242  through the ground and a single one is touching it.
243 
244  It is assumed that update_quantities has been called.
245 
246  @param robot The jiminy robot
247  @param ground_profile The ground profile callback
248  @param position Joint position vector
249  """
250 
251  if use_theoretical_model:
252  pnc_data = robot.pinocchio_data_th
253  else:
254  pnc_data = robot.pinocchio_data
255 
256  # Compute the transform in the world of the contact points
257  contact_frames_transform = []
258  for frame_idx in robot.contact_frames_idx:
259  transform = pnc_data.oMf[frame_idx]
260  contact_frames_transform.append(transform)
261 
262  # Compute the transform of the ground at these points
263  if ground_profile is not None:
264  contact_ground_transform = []
265  ground_pos = np.zeros(3)
266  for frame_transform in contact_frames_transform:
267  ground_pos[2], normal = ground_profile(frame_transform.translation)
268  ground_rot = pin.Quaternion.FromTwoVectors(np.array([0.0, 0.0, 1.0]), normal).matrix()
269  contact_ground_transform.append(pin.SE3(ground_rot, ground_pos))
270  else:
271  contact_ground_transform = len(contact_frames_transform) * [pin.SE3.Identity()]
272 
273  # Compute the position and normal of the contact points wrt their respective ground transform
274  contact_frames_pos_rel = []
275  contact_frames_normal_rel = []
276  for frame_transform, ground_transform in \
277  zip(contact_frames_transform, contact_ground_transform):
278  transform_rel = ground_transform.actInv(frame_transform)
279  contact_frames_pos_rel.append(transform_rel.translation)
280  contact_frames_normal_rel.append(transform_rel.rotation[:, 2])
281 
282  # Compute the closest contact points to the ground in their respective local frame
283  for i in range(len(robot.contact_frames_idx)):
284  height_frame = contact_frames_pos_rel[i] @ contact_frames_normal_rel[i]
285  is_closest = True
286  for j in range(i+1, len(robot.contact_frames_idx)):
287  height = contact_frames_pos_rel[j] @ contact_frames_normal_rel[i]
288  if (height_frame > height + 1e-6): # Add a small 1um tol since "closest" is meaningless at this point
289  is_closest = False
290  break
291  if is_closest:
292  break
293 
294  return robot.contact_frames_names[i]

296 def compute_freeflyer_state_from_fixed_body(robot, position, velocity=None, acceleration=None,
297  fixed_body_name=None, ground_profile=None,
298  use_theoretical_model=True):
299  """
300  @brief Fill rootjoint data from articular data when a body is fixed parallel to world.
301 
302  @details If 'fixed_body_name' is omitted, it will default to the contact point that ensures
303  no contact points are going through the ground and a single one is touching it.
304 
305  @remark The hypothesis is that 'fixed_body_name' is fixed parallel to world frame.
306  So this method computes the position of freeflyer rootjoint in the fixed body frame.
307 
308  @note This function modifies internal data.
309 
310  @param robot The jiminy robot
311  @param[inout] position Must contain current articular data. The rootjoint data can
312  contain any value, it will be ignored and replaced.
313  The method fills in rootjoint data.
314  @param[inout] velocity See position
315  @param[inout] acceleration See position
316  @param fixed_body_name The name of the body frame that is considered fixed parallel to world frame
317  @param ground_profile The ground profile callback
318  """
319  if not robot.has_freeflyer:
320  raise ValueError("The robot does not have a freeflyer.")
321 
322  if use_theoretical_model:
323  pnc_model = robot.pinocchio_model_th
324  pnc_data = robot.pinocchio_data_th
325  else:
326  pnc_model = robot.pinocchio_model
327  pnc_data = robot.pinocchio_data
328 
329  position[:6].fill(0.0)
330  position[6] = 1.0
331  if velocity is not None and acceleration is not None:
332  velocity[:6].fill(0.0)
333  acceleration[:6].fill(0.0)
334  pin.forwardKinematics(pnc_model, pnc_data, position, velocity, acceleration)
335  elif velocity is not None:
336  velocity[:6].fill(0.0)
337  pin.forwardKinematics(pnc_model, pnc_data, position, velocity)
338  else:
339  pin.forwardKinematics(pnc_model, pnc_data, position)
340  pin.framesForwardKinematics(pnc_model, pnc_data, position)
341 
342  if fixed_body_name is None:
343  fixed_body_name = _compute_closest_contact_frame(
344  robot, ground_profile, use_theoretical_model)
345 
346  ff_M_fixed_body = get_body_world_transform(
347  robot, fixed_body_name, use_theoretical_model, copy=False)
348 
349  if ground_profile is not None:
350  ground_translation = np.zeros(3)
351  ground_translation[2], normal = ground_profile(ff_M_fixed_body.translation)
352  ground_rotation = pin.Quaternion.FromTwoVectors(np.array([0.0, 0.0, 1.0]), normal).matrix()
353  w_M_ground = pin.SE3(ground_rotation, ground_translation)
354  else:
355  w_M_ground = pin.SE3.Identity()
356 
357  w_M_ff = w_M_ground.act(ff_M_fixed_body.inverse())
358  position[:7] = pin.se3ToXYZQUAT(w_M_ff)
359 
360  if velocity is not None:
361  ff_v_fixed_body = get_body_world_velocity(
362  robot, fixed_body_name, use_theoretical_model)
363  base_link_velocity = - ff_v_fixed_body
364  velocity[:6] = base_link_velocity.vector
365 
366  if acceleration is not None:
367  ff_a_fixedBody = get_body_world_acceleration(
368  robot, fixed_body_name, use_theoretical_model)
369  base_link_acceleration = - ff_a_fixedBody
370  acceleration[:6] = base_link_acceleration.vector
371 
372  return fixed_body_name

374 def compute_efforts_from_fixed_body(robot, position, velocity, acceleration,
375  fixed_body_name, use_theoretical_model=True):
376  """
377  @brief Compute the efforts using RNEA method.
378 
379  @note This function modifies internal data.
380 
381  @param robot The jiminy robot
382  @param position Joint position vector
383  @param velocity See position
384  @param acceleration See position
385  @param fixed_body_name The name of the body frame on which to apply the external forces
386  """
387  if use_theoretical_model:
388  pnc_model = robot.pinocchio_model_th
389  pnc_data = robot.pinocchio_data_th
390  else:
391  pnc_model = robot.pinocchio_model
392  pnc_data = robot.pinocchio_data
393 
394  # Apply a first run of rnea without explicit external forces
395  pin.computeJointJacobians(pnc_model, pnc_data, position)
396  pin.rnea(pnc_model, pnc_data, position, velocity, acceleration)
397 
398  # Initialize vector of exterior forces to zero
399  f_ext = pin.StdVec_Force()
400  f_ext.extend(len(pnc_model.names) * (pin.Force.Zero(),))
401 
402  # Compute the force at the contact frame
403  support_foot_idx = pnc_model.frames[pnc_model.getBodyId(fixed_body_name)].parent
404  f_ext[support_foot_idx] = pnc_data.oMi[support_foot_idx] \
405  .actInv(pnc_data.oMi[1]).act(pnc_data.f[1])
406 
407  # Recompute the efforts with RNEA and the correct external forces
408  tau = pin.rnea(pnc_model, pnc_data, position, velocity, acceleration, f_ext)
409  f_ext = f_ext[support_foot_idx]
410 
411  return tau, f_ext

417 def retrieve_freeflyer(trajectory_data, freeflyer_continuity=True):
418  """
419  @brief Retrieves the freeflyer positions and velocities.
420  The reference frame is the support foot.
421 
422  @param trajectory_data Sequence of States for which to retrieve the freeflyer.
423  """
424  robot = trajectory_data['robot']
425  use_theoretical_model = trajectory_data['use_theoretical_model']
426 
427  contact_frame_prev = None
428  w_M_ff_offset = pin.SE3.Identity()
429  w_M_ff_prev = None
430  for s in trajectory_data['evolution_robot']:
431  # Compute freeflyer using contact frame as reference frame
432  s.contact_frame = compute_freeflyer_state_from_fixed_body(
433  robot, s.q, s.v, s.a, s.contact_frame,
434  None, use_theoretical_model)
435 
436  # Move freeflyer to ensure continuity over time, if requested
437  if freeflyer_continuity:
438  # Extract the current freeflyer transform
439  w_M_ff = pin.XYZQUATToSe3(s.q[:7])
440 
441  # Update the internal buffer of the freeflyer transform
442  if contact_frame_prev is not None \
443  and contact_frame_prev != s.contact_frame:
444  w_M_ff_offset = w_M_ff_offset * w_M_ff_prev * w_M_ff.inverse()
445  contact_frame_prev = s.contact_frame
446  w_M_ff_prev = w_M_ff
447 
448  # Add the appropriate offset to the freeflyer
449  w_M_ff = w_M_ff_offset * w_M_ff
450  s.q[:7] = pin.se3ToXYZQUAT(w_M_ff)

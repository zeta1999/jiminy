22 class EngineAsynchronous:
23  """
24  @brief Wrapper of Jiminy enabling to update of the command and run simulation
25  steps asynchronously. Convenient helper methods are available to set the
26  seed of the simulation, reset it, and display it.
27 
28  @details The method `action` is used to update the command, which
29  is kept in memory until `action` is called again. On its side, the method
30  `step` without argument is used to run simulation steps. The method `step`
31  has an optional argument `dt_desired` to specify the number of simulation
32  steps to perform at once.
33 
34  @remark This class can be used for synchronous purpose. In such a case, one has
35  to call the method `step` specifying the optional argument `action_next`.
36  """
37  def __init__(self, robot, controller=None, engine=None, use_theoretical_model=False, viewer_backend=None):
38  """
39  @brief Constructor
40 
41  @param[in] robot Jiminy robot properly setup (eg sensors already added)
42  @param[in] use_theoretical_model Whether the state corresponds to the theoretical model
43  @param[in] viewer_backend Backend of the viewer (gepetto-gui or meshcat)
44 
45  @return Instance of the wrapper.
46  """
47  # Backup the user arguments
48  self.robot = robot # For convenience, since the engine will manage its lifetime anyway
49  self.use_theoretical_model = use_theoretical_model
50  self.viewer_backend = viewer_backend
51 
52  # Initialize some internal buffers
53  self._t = -1
54  self._state = None
55  self._sensors_data = None
56  self._action = np.zeros((robot.nmotors,))
57 
58  # Instantiate the Jiminy controller if necessary, then initialize it
59  if controller is None:
60  self._controller = jiminy.ControllerFunctor(self._send_command, self._internal_dynamics)
61  else:
62  self._controller = controller
63  self._controller.initialize(robot)
64 
65  # Instantiate the Jiminy engine
66  if controller is None:
67  self.engine = jiminy.Engine()
68  else:
69  self.engine = engine
70  self.engine.initialize(robot, self._controller)
71 
72 
73  self._viewer = None
74  self._is_viewer_available = False
75 
76 
77  self.step_dt_prev = -1
78 
79  # Reset the low-level jiminy engine
80  self.engine.reset()

83  """
84  @brief This method implement the callback function required by Jiminy

88 
89  @remark This is a hidden function that is not listed as part of the
90  member methods of the class. It is not intended to be called

94  uCommand[:] = self._action

106  pass

108  def seed(self, seed):
109  """
110  @brief Set the seed of the simulation and reset the simulation.
111 
112  @details Note that it also resets the low-level jiminy Engine.
113  One must call the `reset` method manually afterward.
114 
115  @param[in] seed Desired seed (Unsigned integer 32 bits)
116  """
117  assert isinstance(seed, np.uint32), "'seed' must have type np.uint32."
118 
119  engine_options = self.engine.get_options()
120  engine_options["stepper"]["randomSeed"] = np.array(seed, dtype=np.dtype('uint32'))
121  self.engine.set_options(engine_options)
122  self.engine.reset()

124  def reset(self, x0, is_state_theoretical=None):
125  """
126  @brief Reset the simulation.
127 
128  @remark It does NOT start the simulation immediately but rather wait for the
129  first 'step' call. At this point, the sensors data are zeroed, until
130  the simulation actually starts.
131  Note that once the simulations starts, it is no longer possible to
132  changed the robot (included options).
133 
134  @param[in] x0 Desired initial state
135  @param[in] is_state_theoretical Wether the provided initial state is associated
136  with the theoretical or actual model
137  """
138  # Handling of default argument(s)
139  if is_state_theoretical is None:
140  is_state_theoretical = self.use_theoretical_model
141 
142  # Reset the simulation
143  self.engine.reset()
144 
145  # Call update_quantities in order to the frame placement for rendering
146  if is_state_theoretical:
147  x0_rigid = x0
148  if self.robot.is_flexible:
149  x0 = self.robot.get_flexible_state_from_rigid(x0_rigid)
150  else:
151  if self.robot.is_flexible and self.use_theoretical_model:
152  x0_rigid = self.robot.get_rigid_state_from_flexible(x0)
153 
154  # Start the engine, in order to initialize the sensors data
155  self._state = x0_rigid if self.use_theoretical_model else x0
156  self.engine.start(self._state, self.use_theoretical_model)
157 
158  # Backup the sensor data by doing a deep copy manually
159  self._sensors_data = jiminy.sensorsData({
160  _type: {
161  name: self.robot.sensors_data[_type, name].copy()
162  for name in self.robot.sensors_data.keys(_type)
163  }
164  for _type in self.robot.sensors_data.keys()
165  })
166 
167  # Initialize some internal buffers
168  self._t = 0.0
169  self.step_dt_prev = -1
170 
171  # Stop the engine, to avoid locking the robot and the telemetry
172  # too early, so that it remains possible to register external
173  # forces, register log variables, change the options...etc.
174  self.engine.reset()

176  def step(self, action_next=None, dt_desired=-1):
177  """
178  @brief Run simulation steps.
179 
180  @details Even if Jiminy Engine performs several simulation steps
181  internally, this method only output the final state.
182 
183  @param[in] action_next Updated command
184  Optional: Use the value in the internal buffer otherwise
185  @param[in] dt_desired Simulation time difference between before and after the steps.
186  Optional: Perform a single integration step otherwise
187 
188  @return Final state of the simulation
189  """
190  if (not self.engine.is_simulation_running):
191  flag = self.engine.start(self._state, self.use_theoretical_model)
192  if (flag != jiminy.hresult_t.SUCCESS):
193  raise ValueError("Failed to start the simulation.")
194 
195  if (action_next is not None):
196  self.action = action_next
197 
198  return_code = self.engine.step(dt_desired)
199  if (return_code != jiminy.hresult_t.SUCCESS):
200  raise ValueError("Failed to perform the simulation step.")
201 
202  self._t = self.engine.stepper_state.t
203  self._state = None # Do not fetch the new current state if not requested to the sake of efficiency
204  self.step_dt_prev = self.engine.stepper_state.dt

206  def render(self, return_rgb_array=False):
207  """
208  @brief Render the current state of the simulation. One can display it
209  in Gepetto-viewer or return an RGB array.
210 
211  @remark Note that it supports parallel rendering, which means that one
212  can display multiple simulations in the same Gepetto-viewer
213  processes at the same time in different tabs.
214  Note that returning an RGB array required Gepetto-viewer.
215 
216  @param[in] return_rgb_array Updated command
217  Optional: Use the value in the internal buffer otherwise
218 
219  @return Low-resolution rendering as an RGB array (3D numpy array)
220  """
221  rgb_array = None
222 
223  try:
224  # Instantiate the robot and viewer client if necessary
225  if (self._viewer is None):
226  uniq_id = next(tempfile._get_candidate_names())
227  self._viewer = Viewer(self.robot,
228  use_theoretical_model=False,
229  backend=self.viewer_backend,
230  robot_name="_".join(("robot", uniq_id)),
231  scene_name="_".join(("scene", uniq_id)),
232  window_name="_".join(("window", uniq_id)))
233  self._viewer.setCameraTransform(translation=[0.0, 9.0, 2e-5],
234  rotation=[np.pi/2, 0.0, np.pi])
235 
236  # Refresh viewer
237  self._viewer.refresh()
238  self._is_viewer_available = True
239 
240  # Compute rgb array if needed
241  if return_rgb_array:
242  rgb_array = self._viewer.captureFrame()
243  except:
244  self.close()
245  self._viewer = None
246  if self._is_viewer_available:
247  self._is_viewer_available = False
248  return self.render(return_rgb_array)
249  finally:
250  return rgb_array

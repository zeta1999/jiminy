18 def is_log_binary(filename):
19  """
20  @brief Return True if the given filename appears to be binary log file.
21 
22  @details File is considered to be binary log if it contains a NULL byte.
23  From https://stackoverflow.com/a/11301631/4820605.
24  """
25  with open(filename, 'rb') as f:
26  for block in f:
27  if b'\0' in block:
28  return True
29  return False

31 def read_log(filename):
32  """
33  Read a logfile from jiminy. This function supports both text (csv)
34  and binary log.
35 
36  Parameters:
37  - filename: Name of the file to load.
38  Retunrs:
39  - A dictionnary containing the logged values, and a dictionnary
40  containing the constants.
41  """
42 
43  if is_log_binary(filename):
44  # Read binary file using C++ parser.
45  data_dict, constants_dict = Engine.read_log_binary(filename)
46  else:
47  # Read text csv file.
48  constants_dict = {}
49  with open(filename, 'r') as log: constants_str = next(log).split(', ')
50  for c in constants_str:
51  c_split = c.split('=')
52  # Remove line end for last constant.
53  constants_dict[c_split[0]] = c_split[1].strip('\n')
54  # Read data from the log file, skipping the first line (the constants).
55  data = {}
56  reader = DictReader(log)
57  for key, value in reader.__next__().items():
58  data[key] = [value]
59  for row in reader:
60  for key, value in row.items():
61  data[key].append(value)
62  for key, value in data.items():
63  data[key] = np.array(value, dtype=np.float64)
64  # Convert every element to array to provide same API as the C++ parser,
65  # removing spaces present before the keys.
66  data_dict = {k.strip() : np.array(v) for k,v in data.items()}
67  return data_dict, constants_dict

69 def plot_log():
70  description_str = \
71  "Plot data from a jiminy log file using matplotlib.\n" + \
72  "Specify a list of fields to plot, separated by a colon for plotting on the same subplot.\n\n" + \
73  "Example: h1 h2:h3:h4 generates two subplots, one with h1, one with h2, h3, and h4.\n" + \
74  "Wildcard token '*' can be used. In such a case:\n" + \
75  "- If *h2* matches several fields : each field will be plotted individually in subplots. \n" + \
76  "- If :*h2* or :*h2*:*h3*:*h4* matches several fields : each field will be plotted in the same subplot. \n" + \
77  "- If *h2*:*h3*:*h4* matches several fields : each match of h2, h3, and h4 will be plotted jointly in subplots.\n" + \
78  " Note that if the number of matches for h2, h3, h4 differs, only the minimum number will be plotted.\n" + \
79  "\nEnter no plot command (only the file name) to view the list of fields available inside the file."
80 
81  parser = argparse.ArgumentParser(description = description_str, formatter_class = argparse.RawTextHelpFormatter)
82  parser.add_argument("input", help="Input logfile.")
83  parser.add_argument("-c", "--compare", type=str, default=None, help="Colon-separated list of comparison log files:" +\
84  " the same data as the original log will be plotted in the same subplot, with different line styes."+ \
85  " These logfiles must be of the same length and contain the same header as the original log file.\n" +\
86  " Note: you can click on the figure top legend to show / hide data from specific files.")
87  main_arguments, plotting_commands = parser.parse_known_args()
88 
89  # Load log file.
90  log_data, _ = read_log(main_arguments.input)
91 
92  # If no plotting commands, display the list of headers instead.
93  if len(plotting_commands) == 0:
94  print("Available data:")
95  print("\n - ".join(log_data.keys()))
96  exit(0)
97 
98  # Load comparision logs, if any.
99  compare_data = OrderedDict()
100  if main_arguments.compare is not None:
101  for filename in main_arguments.compare.split(':'):
102  compare_data[filename], _ = read_log(filename)
103  # Define linestyle cycle that will be used for comparison logs.
104  linestyles = ["--","-.",":"]
105 
106  # Parse plotting arguments.
107  plotted_elements = []
108  for cmd in plotting_commands:
109  # Check that the command is valid, i.e. that all elements exits. If it is the case, add it to the list.
110  same_subplot = (cmd[0] == ':')
111  headers = cmd.strip(':').split(':')
112 
113  # Expand each element according to wildcard expression.
114  matching_headers = []
115  for h in headers:
116  match = sorted(fnmatch.filter(log_data.keys(), h))
117  if len(match) > 0:
118  matching_headers.append(match)
119  else:
120  print(f"No matching headers for expression {h}")
121  if len(matching_headers) == 0:
122  continue
123 
124  # Compute number of subplots.
125  if same_subplot:
126  plotted_elements.append([e for l_sub in matching_headers for e in l_sub])
127  else:
128  n_subplots = min([len(l) for l in matching_headers])
129  for i in range(n_subplots):
130  plotted_elements.append([l[i] for l in matching_headers])
131 
132  # Create figure.
133  n_plot = len(plotted_elements)
134 
135  if n_plot == 0:
136  print(f"Nothing to plot. Exiting...")
137  return
138 
139  fig = plt.figure()
140 
141  # Create subplots, arranging them in a rectangular fashion.
142  # Do not allow for n_cols to be more than n_rows + 2
143  n_cols = n_plot
144  n_rows = 1
145  while n_cols > n_rows + 2:
146  n_rows = n_rows + 1
147  n_cols = np.ceil(n_plot / (1.0 * n_rows))
148 
149  axs = []
150  for i in range(n_plot):
151  ax = fig.add_subplot(int(n_rows), int(n_cols), i+1)
152  if i > 0:
153  ax.get_shared_x_axes().join(axs[0], ax)
154  axs.append(ax)
155 
156  # Store lines in dictionnary fine_name -> plotted lines, to toggle visibility.
157  main_name = os.path.basename(main_arguments.input)
158  plotted_lines = {main_name : []}
159  for c in compare_data:
160  plotted_lines[os.path.basename(c)] = []
161 
162  plt.gcf().canvas.set_window_title(main_arguments.input)
163  t = log_data['Global.Time']
164  # Plot each element.
165  for ax, plotted_elem in zip(axs, plotted_elements):
166  for name in plotted_elem:
167  line = ax.plot(t, log_data[name], label = name)
168  plotted_lines[main_name].append(line[0])
169 
170  linecycler = cycle(linestyles)
171  for c in compare_data:
172  l = ax.plot(compare_data[c]['Global.Time'], compare_data[c][name], next(linecycler), color=line[0].get_color())
173  plotted_lines[os.path.basename(c)].append(l[0])
174 
175  # Add legend and grid for each plot.
176  for ax in axs:
177  ax.set_xlabel('time (s)')
178  ax.legend()
179  ax.grid()
180 
181  # If a compare plot is present, add overall legend specifying line types.
182  plt.subplots_adjust(bottom=0.05, top=0.98, left=0.06, right=0.98, wspace=0.1, hspace=0.12)
183  if len(compare_data) > 0:
184  linecycler = cycle(linestyles)
185 
186  # Dictionnary: line in legend to log name.
187  legend_lines = {Line2D([0], [0], color='k') : main_name}
188  for c in compare_data:
189  legend_lines[Line2D([0], [0], color='k', linestyle=next(linecycler))] = os.path.basename(c)
190 
191  legend = fig.legend(legend_lines.keys(), legend_lines.values(), loc = 'upper center', ncol = 3)
192  # Create a map (picker) -> (log name) for both the lines and the legend text.
193  picker_to_name = {}
194  for legline, name in zip(legend.get_lines(), legend_lines.values()):
195  legline.set_picker(10) # 10 pts tolerance
196  picker_to_name[legline] = name
197  for legline, name in zip(legend.get_texts(), legend_lines.values()):
198  legline.set_picker(10) # 10 pts tolerance
199  picker_to_name[legline] = name
200 
201  # Increase top margin to fit legend
202  fig.canvas.draw()
203  legend_height = legend.get_window_extent().inverse_transformed(fig.transFigure).height
204  plt.subplots_adjust(top = 0.98 - legend_height)

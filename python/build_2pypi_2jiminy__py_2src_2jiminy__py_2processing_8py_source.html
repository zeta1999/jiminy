10 def smoothing_filter(time_in, val_in, time_out=None, relabel=None, params=None):
11  """
12  @brief Smoothing filter with relabeling and resampling features.
13 
14  @details It supports evenly sampled multidimensional input signal.
15  Relabeling can be used to infer the value of samples at
16  time steps before and after the explicitly provided samples.
17  As a reminder, relabeling is a generalization of periodicity.
18 
19  @param[in] time_in Time steps of the input signal (1D numpy array)
20  @param[in] val_in Sampled values of the input signal
21  (2D numpy array: row = sample, column = time)
22  @param[in] time_out Time steps of the output signal (1D numpy array)
23  @param[in] relabel Relabeling matrix (identity for periodic signals)
24  Optional: Disable if omitted
25  @param[in] params Parameters of the filter. Dictionary with keys:
26  'mixing_ratio_1': Relative time at the begining of the signal
27  during the output signal corresponds to a
28  linear mixing over time of the filtered and
29  original signal. (only used if relabel is omitted)
30  'mixing_ratio_2': Relative time at the end of the signal
31  during the output signal corresponds to a
32  linear mixing over time of the filtered and
33  original signal. (only used if relabel is omitted)
34  'smoothness'[0]: Smoothing factor to filter the begining of the signal
35  (only used if relabel is omitted)
36  'smoothness'[1]: Smoothing factor to filter the end of the signal
37  (only used if relabel is omitted)
38  'smoothness'[2]: Smoothing factor to filter the middle part of the signal
39 
40  @return Filtered signal (2D numpy array: row = sample, column = time)
41  """
42  if time_out is None:
43  time_out = time_in
44  if params is None:
45  params = dict()
46  params['mixing_ratio_1'] = 0.12
47  params['mixing_ratio_2'] = 0.04
48  params['smoothness'] = [0.0,0.0,0.0]
49  params['smoothness'][0] = 5e-3
50  params['smoothness'][1] = 5e-3
51  params['smoothness'][2] = 3e-3
52 
53  if relabel is None:
54  mix_fit = [None,None,None]
55  mix_fit[0] = lambda t: 0.5*(1+np.sin(1/params['mixing_ratio_1']*((t-time_in[0])/(time_in[-1]-time_in[0]))*np.pi-np.pi/2))
56  mix_fit[1] = lambda t: 0.5*(1+np.sin(1/params['mixing_ratio_2']*((t-(1-params['mixing_ratio_2'])*time_in[-1])/(time_in[-1]-time_in[0]))*np.pi+np.pi/2))
57  mix_fit[2] = lambda t: 1
58 
59  val_fit = []
60  for jj in range(val_in.shape[0]):
61  val_fit_jj = []
62  for kk in range(len(params['smoothness'])):
63  val_fit_jj.append(UnivariateSpline(time_in, val_in[jj], s=params['smoothness'][kk]))
64  val_fit.append(val_fit_jj)
65 
66  time_out_mixing = [None, None, None]
67  time_out_mixing_ind = [None, None, None]
68  time_out_mixing_ind[0] = time_out < time_out[-1]*params['mixing_ratio_1']
69  time_out_mixing[0] = time_out[time_out_mixing_ind[0]]
70  time_out_mixing_ind[1] = time_out > time_out[-1]*(1-params['mixing_ratio_2'])
71  time_out_mixing[1] = time_out[time_out_mixing_ind[1]]
72  time_out_mixing_ind[2] = np.logical_and(np.logical_not(time_out_mixing_ind[0]), np.logical_not(time_out_mixing_ind[1]))
73  time_out_mixing[2] = time_out[time_out_mixing_ind[2]]
74 
75  val_out = np.zeros((val_in.shape[0],len(time_out)))
76  for jj in range(val_in.shape[0]):
77  for kk in range(len(time_out_mixing)):
78  val_out[jj,time_out_mixing_ind[kk]] = \
79  (1 - mix_fit[kk](time_out_mixing[kk])) * val_fit[jj][kk](time_out_mixing[kk]) + \
80  mix_fit[kk](time_out_mixing[kk]) * val_fit[jj][-1](time_out_mixing[kk])
81  else:
82  time_tmp = np.concatenate([time_in[:-1]-time_in[-1],time_in,time_in[1:]+time_in[-1]])
83  val_in_tmp = np.concatenate([relabel.dot(val_in[:,:-1]),val_in,relabel.dot(val_in[:,1:])], axis=1)
84  val_out = np.zeros((val_in.shape[0],len(time_out)))
85  for jj in range(val_in_tmp.shape[0]):
86  f = UnivariateSpline(time_tmp, val_in_tmp[jj], s=params['smoothness'][-1])
87  val_out[jj] = f(time_out)
88 
89  return val_out

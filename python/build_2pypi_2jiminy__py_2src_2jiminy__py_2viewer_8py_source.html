291  @staticmethod
292  def _get_colorized_urdf(urdf_path, rgb, root_path=None):
293  """
294  @brief Generate a unique colorized URDF.
295 
296  @remark Multiple identical URDF model of different colors can be
297  loaded in Gepetto-viewer this way.
298 
299  @param[in] urdf_path Full path of the URDF file
300  @param[in] rgb RGB code defining the color of the model. It is the same for each link.
301  @param[in] root_path Root path of the meshes.
302 
303  @return Full path of the colorized URDF file.
304  """
305 
306  color_string = "%.3f_%.3f_%.3f_1.0" % rgb
307  color_tag = "<color rgba=\"%.3f %.3f %.3f 1.0\"" % rgb # don't close tag with '>', in order to handle <color/> and <color></color>
308  colorized_tmp_path = os.path.join(tempfile.gettempdir(), "colorized_urdf_rgba_" + color_string)
309  colorized_urdf_path = os.path.join(colorized_tmp_path, os.path.basename(urdf_path))
310  if not os.path.exists(colorized_tmp_path):
311  os.makedirs(colorized_tmp_path)
312 
313  with open(urdf_path, 'r') as urdf_file: colorized_contents = urdf_file.read()
314 
315  for mesh_fullpath in re.findall('<mesh filename="(.*)"', colorized_contents):
316  # Replace package path by root_path.
317  if mesh_fullpath.startswith('package://'):
318  mesh_fullpath = root_path + mesh_fullpath[len("package:/"):]
319  colorized_mesh_fullpath = os.path.join(colorized_tmp_path, mesh_fullpath[1:])
320  colorized_mesh_path = os.path.dirname(colorized_mesh_fullpath)
321  if not os.access(colorized_mesh_path, os.F_OK):
322  os.makedirs(colorized_mesh_path)
323  shutil.copy2(mesh_fullpath, colorized_mesh_fullpath)
324  colorized_contents = colorized_contents.replace('"' + mesh_fullpath + '"',
325  '"' + colorized_mesh_fullpath + '"', 1)
326  colorized_contents = re.sub("<color rgba=\"[\d. ]*\"", color_tag, colorized_contents)
327 
328  with open(colorized_urdf_path, 'w') as colorized_urdf_file:
329  colorized_urdf_file.write(colorized_contents)
330 
331  return colorized_urdf_path

434  # translation : [Px, Py, Pz],

506  def display(self, evolution_robot, replay_speed, xyz_offset=None):
507  t = [s.t for s in evolution_robot]
508  i = 0
509  init_time = time.time()
510  while i < len(evolution_robot):
511  s = evolution_robot[i]
512  if (xyz_offset is not None):
513  q = s.q.copy() # Make sure to use a copy to avoid altering the original data
514  q[:3] += xyz_offset
515  else:
516  q = s.q
517  try:
518  with self._lock:
519  self._rb.display(q)
520  except Viewer._backend_exception:
521  break
522  t_simu = (time.time() - init_time) * replay_speed
523  i = bisect_right(t, t_simu)
524  sleep(s.t - t_simu)

527 def extract_viewer_data_from_log(log_data, robot):
528  """
529  @brief Extract the minimal required information from raw log data in
530  order to replay the simulation in a viewer.
531 
532  @details It extracts the time and joint positions evolution.
533  .
534  @remark Note that the quaternion angular velocity vectors are expressed
535  it body frame rather than world frame.
536 
537  @param[in] log_data Data from the log file, in a dictionnary.
538  @param[in] robot Jiminy robot.
539 
540  @return Trajectory dictionary. The actual trajectory corresponds to
541  the field "evolution_robot" and it is a list of State object.
542  The other fields are additional information.
543  """
544 
545  # Get the current robot model options
546  model_options = robot.get_model_options()
547 
548  # Extract the joint positions time evolution
549  t = log_data["Global.Time"]
550  try:
551  qe = np.stack([log_data["HighLevelController." + s]
552  for s in robot.logfile_position_headers], axis=-1)
553  except:
554  model_options['dynamics']['enableFlexibleModel'] = not robot.is_flexible
555  robot.set_model_options(model_options)
556  qe = np.stack([log_data["HighLevelController." + s]
557  for s in robot.logfile_position_headers], axis=-1)
558 
559  # Determine whether the theoretical model of the flexible one must be used
560  use_theoretical_model = not robot.is_flexible
561 
562  # Make sure that the flexibilities are enabled
563  model_options['dynamics']['enableFlexibleModel'] = True
564  robot.set_model_options(model_options)
565 
566  # Create state sequence
567  evolution_robot = []
568  for i in range(len(t)):
569  evolution_robot.append(State(t=t[i], q=qe[i]))
570 
571  return {'evolution_robot': evolution_robot,
572  'robot': robot,
573  'use_theoretical_model': use_theoretical_model}

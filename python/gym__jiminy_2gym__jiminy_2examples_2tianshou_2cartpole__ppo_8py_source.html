89 class Actor(nn.Module):
90  def __init__(self, preprocess_net, action_space, free_log_std=True):
91  super().__init__()
92  self.preprocess = preprocess_net
93  self.is_action_space_discrete = isinstance(action_space, gym.spaces.Discrete)
94  self.free_log_std = free_log_std
95  if self.is_action_space_discrete:
96  n_output = np.prod(action_space.n)
97  self.onehot = nn.Linear(64, n_output)
98  else:
99  n_output = np.prod(action_space.shape)
100  self.mu = nn.Linear(64, n_output)
101  if self.free_log_std:
102  self.sigma = nn.Parameter(torch.zeros(n_output, 1))
103  else:
104  self.sigma = nn.Linear(64, n_output)
105 
106  def forward(self, obs, state=None, info={}):
107  logits, h = self.preprocess(obs, state)
108  if self.is_action_space_discrete:
109  logits = nn.functional.softmax(self.onehot(logits), dim=-1)
110  return logits, h
111  else:
112  mu = self.mu(logits)
113  if self.free_log_std:
114  sigma = self.sigma.expand(mu.shape).exp()
115  else:
116  sigma = torch.exp(self.sigma(logits))
117  return (mu, sigma), None

126  logits, _ = self.preprocess(obs, state=kwargs.get('state', None))
127  logits = self.last(logits)

163 def onpolicy_trainer(
164  policy: BasePolicy,
165  train_collector: Collector,
166  test_collector: Collector,
167  max_epoch: int,
168  frame_per_epoch: int,
169  collect_per_step: int,
170  repeat_per_collect: int,
171  episode_per_test: Union[int, List[int]],
172  batch_size: int,
173  train_fn: Optional[Callable[[int], None]] = None,
174  test_fn: Optional[Callable[[int], None]] = None,
175  stop_fn: Optional[Callable[[float], bool]] = None,
176  save_fn: Optional[Callable[[BasePolicy], None]] = None,
177  log_fn: Optional[Callable[[dict], None]] = None,
178  writer: Optional[SummaryWriter] = None,
179  log_interval: int = 1,
180  verbose: bool = True,
181  **kwargs
182 ) -> Dict[str, Union[float, str]]:
183  global_step = 0
184  best_epoch, best_reward = -1, -1
185  stat = {}
186  start_time = time.time()
187  test_in_train = train_collector.policy == policy
188  for epoch in range(1, 1 + max_epoch):
189  # train
190  policy.train()
191  if train_fn:
192  train_fn(epoch)
193  with tqdm.tqdm(total=frame_per_epoch, desc=f'Epoch #{epoch}',
194  **tqdm_config) as t:
195  while t.n < t.total:
196  result = train_collector.collect(n_step=collect_per_step,
197  log_fn=log_fn)
198  data = {}
199  if test_in_train and stop_fn and stop_fn(result['rew']):
200  test_result = test_episode(
201  policy, test_collector, test_fn,
202  epoch, episode_per_test)
203  if stop_fn and stop_fn(test_result['rew']):
204  if save_fn:
205  save_fn(policy)
206  for k in result.keys():
207  data[k] = f'{result[k]:.2f}'
208  t.set_postfix(**data)
209  return gather_info(
210  start_time, train_collector, test_collector,
211  test_result['rew'])
212  else:
213  policy.train()
214  if train_fn:
215  train_fn(epoch)
216  losses = policy.learn(
217  train_collector.sample(0), batch_size, repeat_per_collect)
218  train_collector.reset_buffer()
219  global_step += collect_per_step
220  for k in result.keys():
221  data[k] = f'{result[k]:.2f}'
222  if writer and global_step % log_interval == 0:
223  writer.add_scalar(
224  k, result[k], global_step=global_step)
225  for k in losses.keys():
226  if stat.get(k) is None:
227  stat[k] = MovAvg()
228  stat[k].add(losses[k])
229  data[k] = f'{stat[k].get():.6f}'
230  if writer and global_step % log_interval == 0:
231  writer.add_scalar(
232  k, stat[k].get(), global_step=global_step)
233  t.update(collect_per_step)
234  t.set_postfix(**data)
235  if t.n <= t.total:
236  t.update()
237  # test
238  result = test_episode(
239  policy, test_collector, test_fn, epoch, episode_per_test)
240  if best_epoch == -1 or best_reward < result['rew']:
241  best_reward = result['rew']
242  best_epoch = epoch
243  if save_fn:
244  save_fn(policy)
245  if verbose:
246  print(f'Epoch #{epoch}: test_reward: {result["rew"]:.6f}, '
247  f'best_reward: {best_reward:.6f} in #{best_epoch}')
248  if stop_fn and stop_fn(best_reward):
249  break
250  return gather_info(
251  start_time, train_collector, test_collector, best_reward)

35  k = np.array([100, 20, 50])
36  nu = np.array([0.1, 0.2, 0.2])

38  # Create controllers
39  class Controllers():
40  def __init__(self, k, nu):
41  self.k = k
42  self.nu = nu
43 
44  def compute_command(self, t, q, v, sensors_data, u):
45  u[:] = 0
46

48  u[:] = - self.k * q - self.nu * v
49
